//모달배너 닫기
$('.btn-close').on({
  'click':function(){
    $('.popup-par').css('display','none');
    $('.wrap').css('height','auto');
    if($('.chk-box input').is(':checked')){
      $.cookie('modal-popup', 'no', { expires: 1 });
    }
  }
})

// 만들어진 쿠키를 안보이게 하기
if($.cookie('modal-popup') == 'no'){
  $('.popup-par').css('display','none');
  $('.wrap').css('height','auto');
}else{
  $('.popup-par').css('display','block');
}
</code>
